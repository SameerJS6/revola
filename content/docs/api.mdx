### Anatomy
Import all parts and piece them together.

```tsx 
import {
  ResponsiveDialog, 
  ResponsiveDialogTrigger, 
  ResponsiveDialogContent, 
  ResponsiveDialogHeader, 
  ResponsiveDialogTitle, 
  ResponsiveDialogDescription, 
  ResponsiveDialogFooter
  } from '@/components/ui/revola'
```

```tsx
export function MyRevola() {
  return (
    <ResponsiveDialog>
        <ResponsiveDialogTrigger />
        <ResponsiveDialogContent>
          <ResponsiveDialogHeader>
            <ResponsiveDialogTitle />
            <ResponsiveDialogDescription />
          </ResponsiveDialogHeader>
          <ResponsiveDialogFooter />
        </ResponsiveDialogContent>
    </ResponsiveDialog>
  );
}
```

#### Snap Points (Mobile Drawer Only)
These props configure snap points and only apply when Revola renders as a drawer on mobile devices.

<TypeTable type={{
  snapPoints: {
    description: "Array of numbers from 0 to 1 that corresponds to % of the screen a given snap point should take up. You can also use px values, which doesn't take screen height into account.",
    type: 'array',
    typeDescription: 'Array<number | string>',
    default: ''
  },
  activeSnapPoint: {
    description: "The controlled snap point state.",
    type: 'string | number | null'
  },
  setActiveSnapPoint: {
    description: "Event handler called when the snap point state changes.",
    type: 'function',
    typeDescription: "(snapPoint: string | number | null) => void",
  },
  fadeFromIndex: {
    description: "Index of snapPoint from which the overlay fade should be applied. Defaults to the last step of snapPoints",
    type: 'number',
    default: 'snapPoints.length'
  },
  snapSequentialPoint: {
    description: "Disables velocity-based swiping for snap points. Snap points won't be skipped even with high velocity. Useful when each snap point is equally important.",
    type: 'boolean',
    default: 'false'
  }
}}/>
